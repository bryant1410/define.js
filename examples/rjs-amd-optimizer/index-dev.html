Note:

This page loads one of the DefineJS Module examples which shows the regular modules flow control before buikding the app using r.js to concatenate and minify DefineJS modules. Check out the console to see the logs and the loading process of modules.


src/main.js: 

require(['app'], function (app) {

  if (app.utils.isObject(app.options)) {
    console.log('1. utils.isObject(...) function is working as expected.');
  }

  if (app.utils.isString(app.version)) {
    console.log('2. utils.isString(...) function is working as expected.');
  }

  app.lunch();

});


src/app.js: 

define(['utils'],
  function (utils) {

    var app = {
      version: '0.2.2',
      options: {},
      lunch: function () {
        console.log('App just got lunched!');
      },
      utils: utils
    };

    return app;
  });


src/utils.js: 

define([], function () {

  var utils = {
    isObject: function (o) {
      return o === Object(o);
    },
    isString: function (o) {
      return typeof o === 'string';
    }
  };

  return utils;
});
